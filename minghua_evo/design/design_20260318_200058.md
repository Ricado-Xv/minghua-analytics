## 1. 系统概述

### 1.1 设计目标

本次软件设计基于需求分析文档，旨在解决以下核心问题：

| 问题 | 设计目标 |
|------|---------|
| 动词前缀覆盖不足 | 扩展动词模式库，支持查下、看看、卖了等 |
| 时间词组合不完整 | 支持时间+业务词的组合匹配 |
| 业务词同义词缺失 | 建立同义词配置机制 |
| 水果查询不可用 | 支持具体水果名称动态匹配 |
| 匹配优先级混乱 | 优化正则匹配算法 |

### 1.2 核心特性

1. **三级混合意图识别**（保持不变）
2. **可配置的同义词扩展机制**
3. **动态水果库匹配**
4. **智能优先级评分算法**
5. **规则热更新支持**

### 1.3 设计原则

| 原则 | 说明 |
|------|------|
| **开闭原则** | 对扩展开放，对修改封闭（新增规则无需改代码） |
| **配置优先** | 业务规则尽量外置到配置文件 |
| **渐进增强** | 短期修复 → 中期优化 → 长期架构演进 |
| **向后兼容** | 现有规则平滑迁移 |

    def _regex_match(self, message: str) -> tuple:
        """改进的正则匹配（支持优先级评分）"""
        # 新增：智能优先级评分算法
        return self._enhanced_regex_match(message)
    
    def _enhanced_regex_match(self, message: str) -> tuple:
        """增强版正则匹配"""
        all_matches = []
        
        for intent_type, rule in _COMPILED_RULES.items():
            patterns = rule.get("patterns", [])
            priority = rule.get("priority", 0)
            
            for keyword, pattern, base_confidence in patterns:
                match = re.search(pattern, message)
                if match:
                    # 计算综合得分
                    score = self._calculate_score(
                        message=message,
                        match=match,
                        base_confidence=base_confidence,
                        priority=priority,
                        intent_type=intent_type
                    )
                    
                    params = self._extract_params(message, intent_type)
                    all_matches.append({
                        'intent': intent_type,
                        'score': score,
                        'confidence': score,  # 兼容
                        'params': params,
                        'match_length': len(match.group())
                    })
        
        # 按得分排序，返回最高
        if all_matches:
            all_matches.sort(key=lambda x: x['score'], reverse=True)
            best = all_matches[0]
            return best['intent'], best['score'], best['params']
        
        return None, 0.0, {}
    
    def _calculate_score(self, message, match, base_confidence, priority, intent_type) -> float:
        """
        计算综合得分
        
        得分 = 基础置信度 + 优先级权重 + 长度加分 + 组合加分
        """
        score = base_confidence
        
        # 1. 优先级权重
        score += priority * 0.1
        
        # 2. 长度加分（匹配越长越具体）
        length_ratio = len(match.group()) / len(message)
        score += length_ratio * 0.1
        
        # 3. 组合加分（动词+时间+业务词同时出现）
        has_verb = self._check_verb_match(message)
        has_time = self._check_time_match(message)
        has_business = self._check_business_match(message)
        
        combo_count = sum([has_verb, has_time, has_business])
        if combo_count >= 2:
            score += 0.1 * combo_count
        
        return min(score, 1.0)  # 不超过1.0
    
    def _check_verb_match(self, message: str) -> bool:
        """检查是否包含查询动词"""
        verb_patterns = ['查看', '看看', '查下', '查询', '看看', '给我看看', '帮忙看看']
        return any(v in message for v in verb_patterns)
    
    def _check_time_match(self, message: str) -> bool:
        """检查是否包含时间词"""
        time_patterns = ['上周', '本月', '上月', '这周', '上个月', '\\d+月', '第.*周']
        return any(re.search(p, message) for p in time_patterns)
    
    def _check_business_match(self, message: str) -> bool:
        """检查是否包含业务词"""
        business_words = self.synonym_manager.get_all_business_words()
        return any(w in message for w in business_words)

```python
class SynonymManager:
    """同义词管理器"""
    
    def __init__(self):
        self._load_synonyms()
    
    def _load_synonyms(self):
        """从配置文件加载同义词"""
        # 从 intent_rules.yaml 加载 synonym_groups
        self.synonym_groups = _rules_config.get("synonym_groups", {})
        
        # 构建扁平化的业务词集合
        self._business_words = set()
        for group_name, words in self.synonym_groups.items():
            if 'business' in group_name or 'data' in group_name:
                self._business_words.update(words)
        
        # 添加默认业务词
        default_business = ['数据', '报表', '销售', '营收', '生意', '业绩', '周报', '月报']
        self._business_words.update(default_business)
    
    def get_all_business_words(self) -> set:
        """获取所有业务词"""
        return self._business_words
    
    def get_synonyms(self, group_name: str) -> List[str]:
        """获取指定分组的同义词"""
        return self.synonym_groups.get(group_name, [])
    
    def expand_pattern(self, pattern: str) -> List[str]:
        """
        扩展正则表达式模式
        
        例如: "上月.*销售" → ["上月.*销售", "上月.*数据", "上月.*营收", ...]
        """
        expanded = [pattern]
        
        # 替换占位符
        for group_name, words in self.synonym_groups.items():
            if '{group_name}' in pattern:
                for word in words:
                    expanded.append(pattern.replace(f'{{{group_name}}}', word))
        
        return expanded
```

class FruitNameDetector:
    """水果名称检测器"""
    
    def __init__(self):
        self._load_fruit_names()
    
    def _load_fruit_names(self):
        """从数据源加载水果名称"""
        # 方案A: 静态配置

            "苹果", "香蕉", "橙子", "葡萄", "西瓜", "草莓", 
            "芒果", "梨", "桃子", "猕猴桃", "菠萝", "柚子",
            "火龙果", "荔枝", "龙眼", "石榴", "柿子", "樱桃"
        ]
        
        # 方案B: 从数据源动态加载（可选）
        # self._load_from_data_source()
    
    def _load_from_data_source(self):
        """从数据源动态加载水果名称"""
        # 从 ../fruits.json 加载
        fruits_file = os.path.join(os.path.dirname(__file__), "..", "..", "fruits.json")
        if os.path.exists(fruits_file):
            with open(fruits_file, 'r', encoding='utf-8') as f:
                data = json.load(f)
                self.fruit_names = [f.get('name', f) for f in data]
    
    def detect(self, message: str) -> Optional[str]:
        """
        检测消息中是否包含水果名称
        
        Returns:
            水果名称 or None
        """
        for fruit in self.fruit_names:
            if fruit in message:
                return fruit
        return None
    
    def detect_with_action(self, message: str) -> Optional[dict]:
        """
        检测水果名及销售相关动作
        
        Returns:
            {'fruit': '苹果', 'action': '销售', 'query': '多少'} or None
        """
        fruit = self.detect(message)
        if not fruit:
            return None
        
        # 检测销售相关动作
        action_patterns = [
            (r'卖.*多少', '销量'),
            (r'卖.*好', '销售情况'),
            (r'卖.*差', '销售情况'),
            (r'好卖', '销售热度'),
            (r'销量', '销量'),
            (r'销售', '销售'),
        ]
        
        action = None
        query_type = None
        for pattern, act in action_patterns:
            if re.search(pattern, message):
                action = act
                query_type = pattern
                break
        
        if action:
            return {
                'fruit': fruit,
                'action': action,
                'query': query_type
            }
        
        return None
    
    def get_pattern(self) -> str:
        """获取匹配所有水果名的正则表达式"""
        return f"({'|'.join(self.fruit_names)})"

```python
class IntentRulesLoader:
    """意图规则加载器"""
    
    def __init__(self, config_path: str = None):
        self.config_path = config_path or self._get_default_config_path()
        self._rules = None
        self._last_modified = 0
    
    def _get_default_config_path(self) -> str:
        module_dir = os.path.dirname(os.path.abspath(__file__))
        return os.path.join(os.path.dirname(module_dir), "config", "intent_rules.yaml")
    
    def load(self, force: bool = False) -> dict:
        """加载规则（支持热更新）"""
        current_mtime = os.path.getmtime(self.config_path)
        
        if force or self._rules is None or current_mtime > self._last_modified:
            with open(self.config_path, 'r', encoding='utf-8') as f:
                self._rules = yaml.safe_load(f)
            self._last_modified = current_mtime
        
        return self._rules
    
    def reload(self) -> dict:
        """强制重新加载"""
        return self.load(force=True)
```
